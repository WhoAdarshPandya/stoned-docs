Image


Responsive Image#


src/App.js
import { Image } from "stoned";
import "stoned/dist/index.css";
const App = () => {
return (
<React.Fragment>
<Image
alt="img alt"
onClick={(e) => console.log(e)}
src="https://www.wonderplugin.com/wp-content/uploads/2016/06/blue-grape-hyacinths.jpg"
/>
</React.Fragment>
);
};
export default App;
