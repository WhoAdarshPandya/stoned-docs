# Toast

<br />

## Standard Toast

import { useState } from "react";
import { Toast, Button } from "stoned";
import "stoned/dist/index.css";

export const App = () => {
  const [showToast, setShowToast] = useState(false);
  return (
    <div>
      <Button
        onClick={() => {
          setShowToast((prev) => !prev);
        }}
      >
        show normal toast
      </Button>
      {showToast && (
        <span className="z-plus">
          <Toast
            onClick={(e) => {
              console.log(e);
            }}
            onFinish={() => {
              setShowToast(false);
            }}
            renderCloseButton={true}
            onCloseClick={() => {
              setShowToast(false);
            }}
            autoHideDuration={500}
            renderActionButton={false}
            actionButtonName="close"
            onButtonClick={() => {
              setShowToast(false);
            }}
            message="the standard toast"
            variant="standard"
            anchorOrigin={{ vertical: "top", horizontal: "left" }}
          />
        </span>
      )}
    </div>
  );
};

<App />
