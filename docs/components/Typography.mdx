# Typography

<br />

## Headings

import { Typography } from "stoned";
import "stoned/dist/index.css";

### heading 1

<Typography variant="h1">h1. heading</Typography>

```jsx title="src/App.js"
import { Typography } from "stoned";
import "stoned/dist/index.css";

const App = () => {
  return (
    <React.Fragment>
      <Typography variant="h1">h1. heading</Typography>
    </React.Fragment>
  );
};

export default App;
```

### body2

<Typography
  muted
  variant="body2"
  align="center"
  onClick={(e) => console.log(e)}
>
  this is body2
</Typography>

```jsx title="src/App.js"
import { Typography } from "stoned";
import "stoned/dist/index.css";

const App = () => {
  return (
    <React.Fragment>
      <Typography
        muted
        variant="body2"
        align="center"
        onClick={(e) => console.log(e)}
      >
        this is body2
      </Typography>
    </React.Fragment>
  );
};

export default App;
```
