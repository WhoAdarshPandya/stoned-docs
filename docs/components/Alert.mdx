# Alert

import { Alert } from "stoned";
import "stoned/dist/index.css";

<br />

## Alert Success

<Alert severity="success" />

```jsx title="src/App.js"
import { Alert } from "stoned";
import "stoned/dist/index.css";

const App = () => {
  return (
    <React.Fragment>
      <Alert severity="success" />
    </React.Fragment>
  );
};

export default App;
```
