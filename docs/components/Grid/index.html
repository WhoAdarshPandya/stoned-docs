Single Column#

single col

src/App.js
import { Grid, GridColumn, GridRow, Typography } from "stoned";
import "stoned/dist/index.css";
const App = () => {
return (
<div className="grid_bg">
<Grid>
<GridRow justifyContent="space-between" alignItems="center">
<>
<GridColumn gutter>
<div className="grid_col">
<Typography variant="h6" muted>
single col
</Typography>
</div>
</GridColumn>
</>
</GridRow>
</Grid>
</div>
);
};
export default App;
