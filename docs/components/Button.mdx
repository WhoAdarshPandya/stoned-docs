# Button

<br />

## Contained Buttons

import { Button } from "stoned";
import "stoned/dist/index.css";

<Button color="primary">Primary</Button>
<Button color="secondary">secondary</Button>
<Button color="primary" disableRipple>
  primary (no ripple)
</Button>
<Button color="secondary" disableRipple>
  secondary (no ripple)
</Button>

```jsx title="src/App.js" import {Button} from "stoned"; import "stoned/dist/index.css";
const App = () => {
  return (
    <React.Fragment>
      <Button color="primary">primary</Button>
      <Button color="secondary">secondary</Button>
      <Button color="primary" disableRipple>
        primary
      </Button>
      <Button color="secondary" disableRipple>
        secondary
      </Button>
    </React.Fragment>
  );
};

export default App;
```
