Contained Buttons#

src/App.js
const App = () => {
return (
<React.Fragment>
<Button color="primary">primary</Button>
<Button color="secondary">secondary</Button>
<Button color="primary" disableRipple>
primary
</Button>
<Button color="secondary" disableRipple>
secondary
</Button>
</React.Fragment>
);
};

Outlined Buttons#

src/App.js
import { Button } from "stoned";
import "stoned/dist/index.css";
const App = () => {
return (
<React.Fragment>
<Button color="primary" variant="outlined">
primary
</Button>
<Button color="secondary" variant="outlined">
secondary
</Button>
</React.Fragment>
);
};

Raised Buttons#

src/App.js
import { Button } from "stoned";
import "stoned/dist/index.css";
const App = () => {
return (
<React.Fragment>
<Button variant="contained" raised={true}>
raised primary
</Button>
<Button variant="outlined" color="secondary" raised={true}>
raised secondary
</Button>
</React.Fragment>
);
};

Transparent Buttons#

src/App.js
import { Button } from "stoned";
import "stoned/dist/index.css";
const App = () => {
return (
<React.Fragment>
<Button color="inherit" textColor="textColorPrimary">
Primary
</Button>
<Button color="inherit" textColor="textColorSecondary">
Secondary
</Button>
</React.Fragment>
);
};
